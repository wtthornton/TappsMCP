## Summary

The Claude Agent SDK (`@anthropic-ai/claude-code` for TypeScript,
`anthropic-ai/claude-code` for Python) allows developers to embed Claude Code
sessions programmatically in their own applications. TappsMCP can serve as an
MCP server within Agent SDK sessions via the `mcp_servers` parameter in SDK
`query()` calls, and TappsMCP subagent definitions can be passed programmatically
via the `agents` parameter. This story provides concrete, runnable integration
examples showing how to use TappsMCP as an MCP server in Agent SDK `query()`
calls, how to define TappsMCP subagents programmatically, and how to build a
custom quality-gate CI pipeline using the Agent SDK with TappsMCP as the quality
backend. Both Python and TypeScript examples are provided. The examples are
placed in `examples/agent-sdk/` in the TappsMCP repository with a corresponding
test that verifies the code compiles and — where possible — runs against a mock
MCP server.

### Python basic example

```python
#!/usr/bin/env python3
"""
Basic TappsMCP + Agent SDK example.

Prerequisites:
  pip install anthropic-ai-claude-code  (hypothetical SDK package name)
  export ANTHROPIC_API_KEY=sk-...
  export TAPPS_MCP_PROJECT_ROOT=/path/to/your/project
"""
from __future__ import annotations

import asyncio
import os

# Adjust import path based on actual SDK package name
from claude_code_sdk import query, ClaudeCodeOptions, McpServerConfig


async def run_quality_check(file_path: str) -> str:
    """Run a TappsMCP quick check on a single file via the Agent SDK."""
    options = ClaudeCodeOptions(
        mcp_servers={
            "tapps-mcp": McpServerConfig(
                command="uvx",
                args=["tapps-mcp", "serve"],
                env={
                    "TAPPS_MCP_PROJECT_ROOT": os.environ.get(
                        "TAPPS_MCP_PROJECT_ROOT",
                        os.getcwd(),
                    ),
                },
            )
        },
        allowed_tools=["mcp__tapps-mcp__tapps_quick_check"],
        max_turns=3,
    )

    result_parts = []
    async for message in query(
        prompt=f"Run tapps_quick_check on {file_path} and report the score and top issues.",
        options=options,
    ):
        if hasattr(message, "content"):
            result_parts.append(str(message.content))

    return "\n".join(result_parts)


if __name__ == "__main__":
    import sys
    file_to_check = sys.argv[1] if len(sys.argv) > 1 else "src/tapps_mcp/server.py"
    result = asyncio.run(run_quality_check(file_to_check))
    print(result)
```

### Python CI pipeline example

```python
#!/usr/bin/env python3
"""
TappsMCP CI quality gate pipeline using the Agent SDK.
Exits non-zero if any changed file fails the quality gate.
"""
from __future__ import annotations

import asyncio
import os
import sys

from claude_code_sdk import query, ClaudeCodeOptions, McpServerConfig


async def run_quality_gate(preset: str = "staging") -> bool:
    """
    Run tapps_validate_changed for all changed files.
    Returns True if all files pass, False otherwise.
    """
    options = ClaudeCodeOptions(
        mcp_servers={
            "tapps-mcp": McpServerConfig(
                command="uvx",
                args=["tapps-mcp", "serve"],
                env={"TAPPS_MCP_PROJECT_ROOT": os.getcwd()},
            )
        },
        allowed_tools=[
            "mcp__tapps-mcp__tapps_session_start",
            "mcp__tapps-mcp__tapps_validate_changed",
        ],
        max_turns=5,
    )

    passed = True
    async for message in query(
        prompt=(
            f"Run tapps_session_start, then run tapps_validate_changed with "
            f"preset={preset}. If any files fail, output 'GATE_FAILED'. "
            "If all pass, output 'GATE_PASSED'."
        ),
        options=options,
    ):
        content = str(getattr(message, "content", ""))
        if "GATE_FAILED" in content:
            passed = False

    return passed


if __name__ == "__main__":
    preset = sys.argv[1] if len(sys.argv) > 1 else "staging"
    passed = asyncio.run(run_quality_gate(preset))
    if not passed:
        print("Quality gate FAILED", file=sys.stderr)
        sys.exit(1)
    print("Quality gate PASSED")
```

### Python subagent example

```python
#!/usr/bin/env python3
"""
Demonstrates registering TappsMCP subagents programmatically via the Agent SDK.
"""
from __future__ import annotations

import asyncio
import os
from textwrap import dedent

from claude_code_sdk import query, ClaudeCodeOptions, McpServerConfig, AgentDefinition


TAPPS_REVIEWER_AGENT = AgentDefinition(
    name="tapps-reviewer",
    description=(
        "Use proactively to review code quality, run security scans, and "
        "enforce quality gates after editing Python files."
    ),
    system_prompt=dedent("""
        You are a TappsMCP quality reviewer. When invoked:
        1. Identify which Python files were recently edited
        2. Call mcp__tapps-mcp__tapps_quick_check on each changed file
        3. If any file scores below 70, call mcp__tapps-mcp__tapps_score_file
        4. Summarize findings: file, score, top issues, suggested fixes
    """).strip(),
    tools=["Read", "Glob", "Grep"],
    model="sonnet",
    permission_mode="dontAsk",
)


async def run_with_reviewer_agent(task: str) -> None:
    """Run a task with the tapps-reviewer subagent available."""
    options = ClaudeCodeOptions(
        mcp_servers={
            "tapps-mcp": McpServerConfig(
                command="uvx",
                args=["tapps-mcp", "serve"],
                env={"TAPPS_MCP_PROJECT_ROOT": os.getcwd()},
            )
        },
        agents=[TAPPS_REVIEWER_AGENT],
        max_turns=10,
    )

    async for message in query(prompt=task, options=options):
        print(message)


if __name__ == "__main__":
    asyncio.run(run_with_reviewer_agent(
        "Review the quality of all Python files changed in the last commit."
    ))
```

### TypeScript basic example

```typescript
// basic_quality_check.ts
// Prerequisites: npm install @anthropic-ai/claude-code
// ANTHROPIC_API_KEY and TAPPS_MCP_PROJECT_ROOT must be set

import { query, type ClaudeCodeOptions } from "@anthropic-ai/claude-code";

async function runQualityCheck(filePath: string): Promise<void> {
  const options: ClaudeCodeOptions = {
    mcpServers: {
      "tapps-mcp": {
        command: "uvx",
        args: ["tapps-mcp", "serve"],
        env: {
          TAPPS_MCP_PROJECT_ROOT:
            process.env.TAPPS_MCP_PROJECT_ROOT ?? process.cwd(),
        },
      },
    },
    allowedTools: ["mcp__tapps-mcp__tapps_quick_check"],
    maxTurns: 3,
  };

  for await (const message of query({
    prompt: `Run tapps_quick_check on ${filePath} and report the score and top issues.`,
    options,
  })) {
    if ("content" in message) {
      console.log(message.content);
    }
  }
}

const filePath = process.argv[2] ?? "src/tapps_mcp/server.py";
runQualityCheck(filePath).catch(console.error);
```

### Approach

1. Create the `examples/agent-sdk/` directory tree with Python and TypeScript
   subdirectories.
2. Write the Python examples using the actual Agent SDK package name confirmed
   from the SDK documentation. The import paths in this story use hypothetical
   names — verify against the actual published SDK before finalizing.
3. Write the TypeScript examples using the `@anthropic-ai/claude-code` npm
   package.
4. Add a `tsconfig.json` targeting ES2022 with strict mode.
5. Add a `package.json` with `@anthropic-ai/claude-code` as a dev dependency.
6. Write `examples/agent-sdk/README.md` with clear prerequisites and
   step-by-step instructions.
7. Add a test in `tests/unit/test_agent_sdk_examples.py` that verifies
   Python files compile using `py_compile`.

### SDK package name caveat

The Agent SDK Python package name and import path should be verified against
the official Anthropic documentation before implementation. As of research
(2026-02-21), the SDK is `@anthropic-ai/claude-code` for npm. The Python
equivalent package name should be confirmed from PyPI or the official docs
before writing the import statements in the examples.
